## 1. 관련 테이블 DDL

### tb_subscription_plan (구독 플랜 정의)

```sql
CREATE TABLE tb_subscription_plan (
    ID                          BIGINT       NOT NULL AUTO_INCREMENT,
    PLAN_NM                     VARCHAR(100) NULL,      -- 'Free' | 'Pro' | 'Business' | 'Enterprise'
    PERIOD_GROUP                VARCHAR(20)  NULL,      -- 'FREE' | 'MONTH' | 'YEAR'
    PERIOD_MONTH                INT          NULL,      -- 0 | 1 | 12
    PRICE                       INT          NULL,
    TOTAL_PRICE                 INT          DEFAULT 0,
    -- ▼ 플랜별 기능 제한값 (NULL = 무제한)
    ALARM_KAKAO_DAY_LIMIT       INT NULL,  -- 카카오 알림 일 제한
    BID_RECOMMEND_DAY_LIMIT     INT NULL,  -- 입찰 공고 추천 일 제한
    PRESPEC_RECOMMEND_DAY_LIMIT INT NULL,  -- 사전규격 공고 추천 일 제한
    AI_SUMMARY_DAY_LIMIT        INT NULL,  -- AI 요약 일 제한 (Free 전용)
    AI_SUMMARY_MONTH_LIMIT      INT NULL,  -- AI 요약 월 제한 (Pro/Business)
    AI_SEARCH_DAY_LIMIT         INT NULL,  -- AI 검색 일 제한 (NULL=무제한)
    ...
    PRIMARY KEY (ID)
);
```

| 4  | Business  | YEAR         | 20             | 25                | 5           | NULL           | 200            | NULL          |

### tb_subscription (회원 구독 정보)

```sql
CREATE TABLE tb_subscription (
    ID                BIGINT      NOT NULL AUTO_INCREMENT,
    USER_ID           VARCHAR(50) NOT NULL,
    PLAN_ID           BIGINT      NOT NULL,              -- tb_subscription_plan.ID FK
    STATUS            VARCHAR(20) NOT NULL DEFAULT 'ACTIVE',
    -- STATUS: ACTIVE | PAUSED | CANCELED | EXPIRED | SCHEDULED
    NEXT_BILLING_DATE DATE        NOT NULL,
    -- Free 플랜은 9999-12-31 고정 (무기한)
    STARTED_AT        DATE        NOT NULL,
    ENDED_AT          DATE        NULL,
    PRIMARY KEY (ID)
);
```

---

### tb_subscription_usage_daily (일별 사용량 집계)

```sql
CREATE TABLE tb_subscription_usage_daily (
    USER_ID      VARCHAR(50) NOT NULL,
    FEATURE_TYPE VARCHAR(30) NOT NULL,
    -- 'ALARM_KAKAO' | 'AI_SUMMARY' | 'AI_SEARCH'
    -- 'BID_RECOMMEND' | 'PRESPEC_RECOMMEND'
    USE_DT       DATE        NOT NULL,  -- 날짜별 자동 분리, 별도 리셋 배치 불필요
    USE_COUNT    INT         NOT NULL DEFAULT 0,
    UPDATED_AT   DATETIME    NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    PRIMARY KEY (USER_ID, FEATURE_TYPE, USE_DT),
    KEY idx_usage_user_dt (USER_ID, USE_DT)
);
```

## 2. 유저 플랜 + 제한값 조회 쿼리

```sql
SELECT
    s.USER_ID,
    s.PLAN_ID,
    s.STATUS,
    p.PLAN_NM,
    p.PERIOD_GROUP,
    p.ALARM_KAKAO_DAY_LIMIT,
    p.BID_RECOMMEND_DAY_LIMIT,
    p.PRESPEC_RECOMMEND_DAY_LIMIT,
    p.AI_SUMMARY_DAY_LIMIT,
    p.AI_SUMMARY_MONTH_LIMIT,
    p.AI_SEARCH_DAY_LIMIT
FROM tb_subscription s
JOIN tb_subscription_plan p ON s.PLAN_ID = p.ID
WHERE s.USER_ID = %s
  AND s.STATUS = 'ACTIVE'
LIMIT 1;
```

## 3. 사용량 체크 & 증가 로직

### 3-1. 당일 사용량 조회

```sql
SELECT USE_COUNT
FROM tb_subscription_usage_daily
WHERE USER_ID = %s
  AND FEATURE_TYPE = %s
  AND USE_DT = CURDATE();
-- 행 없으면 USE_COUNT = 0으로 간주
```

### 3-2. 이번 달 누적 사용량 조회 (AI_SUMMARY 월 제한용)

```sql
SELECT COALESCE(SUM(USE_COUNT), 0) AS MONTH_COUNT
FROM tb_subscription_usage_daily
WHERE USER_ID = %s
  AND FEATURE_TYPE = 'AI_SUMMARY'
  AND USE_DT >= DATE_FORMAT(NOW(), '%Y-%m-01');
```

### 3-3. 사용량 증가 (upsert)

```sql
INSERT INTO tb_subscription_usage_daily (USER_ID, FEATURE_TYPE, USE_DT, USE_COUNT)
VALUES (%s, %s, CURDATE(), 1)
ON DUPLICATE KEY UPDATE USE_COUNT = USE_COUNT + 1;
```

## 4. 제한 체크 흐름 (Python 구현 — `bid-sync/services/usage_limiter.py`)

```python
def get_plan_limits(company_id: int, db: Session) -> dict | None:
    """
    기업의 ACTIVE 구독 플랜 제한값 조회.
    구독 없거나 ACTIVE 아니면 None 반환 → 매칭 스킵.
    NULL = 무제한.
    """
    row = db.execute(text("""
        SELECT p.BID_RECOMMEND_DAY_LIMIT,
               p.PRESPEC_RECOMMEND_DAY_LIMIT,
               p.ALARM_KAKAO_DAY_LIMIT,
               p.AI_SUMMARY_DAY_LIMIT,
               p.AI_SUMMARY_MONTH_LIMIT,
               p.AI_SEARCH_DAY_LIMIT
        FROM tb_subscription s
        JOIN tb_subscription_plan p ON s.PLAN_ID = p.ID
        WHERE s.COMPANY_ID = :cid AND s.STATUS = 'ACTIVE'
        LIMIT 1
    """), {"cid": company_id}).fetchone()

    if not row:
        return None  # 구독 없음 → 매칭 스킵

    return {
        "BID_RECOMMEND":     row[0],
        "PRESPEC_RECOMMEND": row[1],
        "ALARM_KAKAO":       row[2],
        "AI_SUMMARY_DAY":    row[3],
        "AI_SUMMARY_MONTH":  row[4],
        "AI_SEARCH":         row[5],
    }


def get_today_usage(company_id: int, feature_type: str, db: Session) -> int:
    """당일 누적 사용량 조회. 행 없으면 0."""
    row = db.execute(text("""
        SELECT USE_COUNT FROM tb_subscription_usage_daily
        WHERE COMPANY_ID = :cid AND FEATURE_TYPE = :ft AND USE_DT = CURDATE()
    """), {"cid": company_id, "ft": feature_type}).fetchone()
    return row[0] if row else 0


def increment_usage(company_id: int, feature_type: str, amount: int, db: Session) -> None:
    """사용량 일괄 증가 (upsert). amount = 이번 배치에서 추가된 건수."""
    if amount <= 0:
        return
    db.execute(text("""
        INSERT INTO tb_subscription_usage_daily (COMPANY_ID, FEATURE_TYPE, USE_DT, USE_COUNT)
        VALUES (:cid, :ft, CURDATE(), :amt)
        ON DUPLICATE KEY UPDATE USE_COUNT = USE_COUNT + :amt
    """), {"cid": company_id, "ft": feature_type, "amt": amount})
```

### match_company() 내 BID_RECOMMEND 적용 패턴

```python
# 1. 루프 시작 전 — 플랜 체크
limits = get_plan_limits(company_id, db)
if limits is None:
    return {"skipped": True, "reason": "no_subscription"}

bid_recommend_limit = limits["BID_RECOMMEND"]  # None = 무제한
today_usage = get_today_usage(company_id, "BID_RECOMMEND", db) if bid_recommend_limit is not None else 0

if bid_recommend_limit is not None and today_usage >= bid_recommend_limit:
    return {"skipped": True, "reason": "limit_exceeded"}

new_recommend_count = 0  # 이번 배치 신규 추천 건수

# 2. 루프 내 — 신규 저장 시 한도 체크
if not existing:
    if bid_recommend_limit is not None and today_usage + new_recommend_count >= bid_recommend_limit:
        continue  # 한도 도달, 이후 신규 추천 중단
    new_recommend_count += 1

# 기존 매핑 업데이트는 카운트 차감 없음

# 3. 루프 종료 후 — 사용량 일괄 증가
increment_usage(company_id, "BID_RECOMMEND", new_recommend_count, db)
```
